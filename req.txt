make

avrdude -c linuxgpio

sudo nano /etc/avrdude_gpio.conf
programmer
  id = "linuxgpio";
  desc = "Use sysfs interface to bitbang GPIO pins";
  type = "linuxgpio";
  reset = 18;
  sck = 11;
  mosi = 10;
  miso = 9;
;


cd ~/GrovePi/Firmware
